    title: "JavaScript для начинающих",
    author: "Иванов И.И.",
    year: 2023,
    pages: 350,
    getInfo: function() {
        return `${this.title}, автор: ${this.author}, ${this.year} год, ${this.pages} стр.`;
    }
};
